private void click_Aceptar(object sender, RoutedEventArgs e)
        {
            displayImage();
            var encoder = new BmpBitmapEncoder();
            int Height = (int)this.GridWindow.ActualHeight;
            int Width = (int)this.GridWindow.ActualWidth;

            RenderTargetBitmap bmp = new RenderTargetBitmap(Width, Height, 96, 96, PixelFormats.Pbgra32);
            bmp.Render(this.GridWindow);
            BitmapFrame frame = BitmapFrame.Create(bmp);
            encoder.Frames.Add(frame);

            
            using (var stream = File.Create("imagenPrueba.bmp"))//https://stackoverflow.com/questions/10667012/getting-downloads-folder-in-c/21953690
            {
                encoder.Save(stream);
            }
            

            
            MessageBox.Show("llego aqui");
        }

        {
            try
            {
                this.GridWindow.Width = Width;
                this.GridWindow.Height = 800;
                //this.stOuter.Width = Width;Poner un canvas entre el grid que ocupa toda la ventana y el contenido
                //this.stOuter.Height = 800;
            }
            catch
            {
            }
        }
